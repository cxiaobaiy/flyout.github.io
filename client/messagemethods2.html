9.获取消息记录 getMessageRecord ​

tips:请在登录后使用

参数：

params {conversationType:C2C|GROUP,size:分页大小,to：对方userId,messageId:用于消息续拉，即从那一条消息开始拉取，比如分也是传第一页的最早的一条消息的Id}

返回值：Promise<Object>

eg：

ecmascript

import {useFxIm} from 'fx-im-web'

const fxIm = useFxIm()
fxIm.getMessageRecord({
    conversationType:'C2C',
    size:20,
    to:1,
    // messageId:'1839345879890023'  第一页不传，第二页开始才传  或者直接指定到某一条消息之后的消息，用于消息跳转等时可用
}).then(res=>{
    console.log('消息列表',res)
})

10.创建自定义消息 createCustomeMessage ​

tips:请在登录后使用

参数：

keystatus字段描述
to:string,number用户的userId 或 群聊ID
conversationId:string会话Id,取自会话列表
conversationType:string会话列表 可选（C2C 单聊，GROUP 群聊）
data:string发送的自定义文本（JSON字符串）
needReadReceipt:Boolean可选是否需要已读回执，默认为false,即不需要已读回执
atUserList:Array可选@ 成员名单，单聊无效
hasRiskContent:Boolean可选消息进行风险标识
description消息描述

返回值：Promise<Object>

eg：

ecmascript

import {useFxIm} from 'fx-im-web'

const fxIm = useFxIm()
fxIm.createCustomeMessage({
    to:1,
    data:JSON.stringify({
        id:1,
        name:2,
        test:3,
        
    }),
    description:'我是着一条自定义消息的标题描述文字',
    conversationType:'C2C',// 可选GROUP发送群聊消息
    conversationId: 'C2CtestUser' 
}).then(res=>{
    console.log(res,'创建的文本消息')
    fxIm.sendMessage(res)
})

推荐使用 createMessage 创建消息后直接上屏渲染 减少等待时间

11.创建图片消息 createImageMessage ​

tips:请在登录后使用

参数：

keystatus字段描述
to:string,number用户的userId 或 群聊ID
conversationId:string会话Id,取自会话列表
conversationType:string会话列表 可选（C2C 单聊，GROUP 群聊）
file:FILE发送的图片
needReadReceipt:Boolean可选是否需要已读回执，默认为false,即不需要已读回执
atUserList:Array可选@ 成员名单，单聊无效
hasRiskContent:Boolean可选消息进行风险标识
onProgress:function可选上传进度反馈

返回值：Promise<Object>

eg：

ecmascript

import {useFxIm} from 'fx-im-web'

const fxIm = useFxIm()
fxIm.createImageMessage({
    to:1,
    file:new File({name:'213.png',}),  // 传入文件
    conversationType:'C2C',// 可选GROUP发送群聊消息
    conversationId: 'C2CtestUser',
    onProgress:(e)=>{
        console.log(e,'这里处理文件上传的进度')
    }
}).then(res=>{
    console.log(res,'创建的文本消息')
    // 消息商品 同时发送
    fxIm.sendMessage(res)
})

推荐使用 createMessage 创建消息后直接上屏渲染 减少等待时间
